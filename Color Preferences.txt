let favoriteColors = ["Black", "Purple", "Grey"]; // Initialize an empty array to store colors

// Use a loop to prompt the user for three colors
for (let i = 0; i < 3; i++) {
  let color = prompt("Enter your favorite color: ");
  favoriteColors.push(color); // Add the color to the array
  console.log("Your favorite colors so far: " + favoriteColors); // Print the updated array
}
