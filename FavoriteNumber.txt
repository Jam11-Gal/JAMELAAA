// Declare the favorite number
const favNumber = 11;

// Initialize a variable to track the guess
let guess;

// Use a while loop to keep prompting the user until they guess correctly
while (true) {
  // Prompt the user for their guess
  guess = parseInt(prompt("Guess my favorite number: "));

  // Check the guess and provide feedback
  if (guess < favNumber) {
    console.log("Too low! Try again.");
  } else if (guess > favNumber) {
    console.log("Too high! Try again.");
  } else {
    console.log("You guessed it! My favorite number is " + favNumber);
    break;
  }
}
